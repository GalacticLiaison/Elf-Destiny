ruler_check_diarchy = {
	desc = "ruler_check_diarchy"
	selection_tooltip = "elf_debug_menu_tooltip"
	
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_divine_spark.dds"
	}

	decision_group_type = debug

	is_shown = {
		is_ai = no
		OR = {
			# has_global_variable = elf_cheats_on
			debug_only = yes
			has_game_rule = elf_dest_debug_menu_turned_on
		}
	}

	ai_goal = no
	ai_will_do = {
		value = 0
	}
	ai_potential = {
		always = no
	}
	ai_check_interval = 0

	effect = {

		random_ruler = {
			limit = {
				has_diarchy_type = aeluran_regency
				is_any_elf_type = no
				NOT = {
					has_relation_friend = root
				}
			}
			save_scope_as = diarchy_ruler
		}

		if = {
			limit = {
				exists = scope:diarchy_ruler
			}
			set_relation_friend = scope:diarchy_ruler
		}
	}
}
