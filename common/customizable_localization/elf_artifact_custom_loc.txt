ElfArtifactArchetypeTitle  = {
	type = character

	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:ravager
		}
		localization_key = archetype_ravager_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:roadmen
		}
		localization_key = archetype_roadmen_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:paladin
		}
		localization_key = archetype_paladin_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:captain
		}
		localization_key = archetype_captain_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:noble
		}
		localization_key = archetype_noble_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:aeluran
		}
		localization_key = archetype_aeluran_title
	}
	text = {
		trigger = {
			exists = scope:artifact_archetype
			scope:artifact_archetype = flag:commoner
		}
		localization_key = archetype_not_found_title
		fallback = yes
	}
}
