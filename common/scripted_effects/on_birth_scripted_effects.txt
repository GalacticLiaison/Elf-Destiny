determine_endless_lord_parent = {
	if = {
		limit = {
			scope:real_father = {
				is_ruler = yes
			}
			scope:mother = {
				is_ruler = no
			}
		}

		scope:real_father = {
			save_scope_as = lord_to_copy
		}
	}
	else_if = {
		limit = {
			scope:real_father = {
				is_ruler = no
			}
			scope:mother = {
				is_ruler = yes
			}
		}

		scope:mother = {
			save_scope_as = lord_to_copy
		}
	}
	else_if = {
		limit = {
			scope:real_father = {
				is_ruler = yes
			}
			scope:mother = {
				is_ruler = yes
			}

		}

		if = {
			limit = {
				scope:real_father.primary_title.tier > scope:mother.primary_title.tier
			}
			scope:real_father = {
				save_scope_as = lord_to_copy
			}
		}
		if = {
			limit = {
				scope:real_father.primary_title.tier < scope:mother.primary_title.tier
			}
			scope:mother = {
				save_scope_as = lord_to_copy
			}
		}
		if = {
			limit = {
				scope:real_father.primary_title.tier = scope:mother.primary_title.tier
			}
			if = {
				limit = {
					scope:mother = {
						is_ai = no
					}
				}
				scope:mother = {
					save_scope_as = lord_to_copy
				}
			}
			else_if = {
				limit = {
					scope:real_father = {
						is_ai = no
					}
				}
				scope:real_father = {
					save_scope_as = lord_to_copy
				}
			}
			else = {
				scope:mother = {
					save_scope_as = lord_to_copy
				}
			}
		}
	}

}

	scope:child = {
		add_trait = endless_lord

		add_trait_xp = { # previous iteration's xp
			trait = endless_lord
			value = endless_lord_parent_track_experience_to_add
		}
		add_trait_xp = { # plus new level for this new child
			trait = endless_lord
			value = 10
		}
	}
}
