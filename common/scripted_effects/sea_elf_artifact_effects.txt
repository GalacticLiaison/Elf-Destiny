create_crown_of_coral_effect = {
	$OWNER$ = { save_scope_as = owner }

	create_artifact = {
		name = crown_of_coral
		description = crown_of_coral_desc
		template = sea_elf_template
		type = helmet
		visuals = atlnatis_crown
		modifier = crown_of_coral_modifier
		rarity = famed
		decaying = no
		save_scope_as = newly_created_sea_elf_artifact
	}
	scope:newly_created_sea_elf_artifact = {
		set_variable = is_crown_of_coral
	}
}
