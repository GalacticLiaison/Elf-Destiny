create_pirate_hat_effect = {
	$OWNER$ = { save_scope_as = owner }

	create_artifact = {	
		name = pirate_hat
		description = pirate_hat_desc
		template = sea_elf_template
		type = helmet
		visuals = pirate_hat
		modifier = pirate_hat_modifier
		rarity = famed
		decaying = no
		save_scope_as = newly_created_artifact
	}
	scope:newly_created_artifact ?= {
		equip_artifact_to_owner_replace = yes
	}
}
