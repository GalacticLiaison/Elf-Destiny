mark_sea_elf_tradition_as_discovered = {
	$TRADITION$ = {
		save_scope_as = tradition
	}

	if = {
		limit = {
			scope:tradition = culture_tradition:tradition_guardians_of_the_abyss
		}
		hidden_effect = {
			add_to_global_variable_list = {
				name = discovered_sea_elf_traditions
				target = flag:tradition_guardians_of_the_abyss
			}
		}
	}
}
