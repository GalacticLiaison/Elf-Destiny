is_any_elf_type = {
    OR = {
        is_any_elf_immortal_type = yes
        is_any_elf_mortal_type = yes
    }
}

is_any_elf_immortal_type = {
    OR = {
        has_trait = elf_blood
        has_trait = elf
        has_trait = high_elf
        has_trait = true_elf
        has_trait = fae
        has_trait = fae_radiant
        has_trait = celestial
        has_trait = seraphic_celestial
        has_trait = eldar
        has_trait = maiar
        has_trait = valar
        has_trait = aratar
    }
}

        has_trait = seraphic_celestial_mortal

}

is_below_top_elf_type = {
    AND = {
        is_any_elf_type = yes
        NOT = {
            has_trait = aratar
        }
        NOT = {
            has_trait = aratar_mortal
        }
    }
}
