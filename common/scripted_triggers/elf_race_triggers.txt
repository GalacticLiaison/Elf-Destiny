        is_elf_blood = yes
        is_elf = yes

        is_true_elf = yes
        is_fae = yes

        is_celestial = yes
        is_seraphic_celestial = yes
        is_eldar = yes
        is_maiar = yes
        is_valar = yes
        is_aratar = yes
    }
}
