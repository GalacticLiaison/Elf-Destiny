	AND = {
		$FAITH$ = { has_doctrine = doctrine_consanguinity_mandatory }
		NOT = { 
			is_close_or_extended_family_of = $CHARACTER$ #[parents, children, siblings, grandparents, grandchildren, cousins, uncles, aunts, nephews, nieces]
		}
	}
}
