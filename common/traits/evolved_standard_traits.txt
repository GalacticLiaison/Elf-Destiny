			men_at_arms_maintenance = -0.25
			martial = 2
			prowess = 1
		}
	}

	name = {
		first_valid = {
			#Legendary adventurer
			triggered_desc = {
				trigger = {
					exists = this
					has_trait_xp = {
						trait = adventurer
						value >= 100
					}
				}
				desc = trait_adventurer_3
			}
			#Famous adventurer
			triggered_desc = {
				trigger = { 
					exists = this
					has_trait_xp = {
						trait = adventurer
						value >= 50
					}
				}
				desc = trait_adventurer_2
			}
			#Eager adventurer
			desc = trait_adventurer_1
		}
	}

	# Loc
	desc = {
		first_valid = {
			#The trait can never go higher than the first level if it's not assigned to a character
			triggered_desc = {
				trigger = { 
					NOT = { exists = this } 
				}
				desc = trait_adventurer_1_desc
			}
			triggered_desc = {
				trigger = { 
					has_trait_xp = {
						trait = adventurer
						value >= 100
					}
				}
				desc = trait_adventurer_3_character_desc
			}
			triggered_desc = {
				trigger = { 
					has_trait_xp = {
						trait = adventurer
						value >= 50
					}
				}
				desc = trait_adventurer_2_character_desc
			}
			desc = trait_adventurer_1_character_desc
		}
	}

	ai_energy = low_positive_ai_value
	ai_boldness = low_positive_ai_value
	ai_honor = low_negative_ai_value
}
