on_game_start_after_lobby = { 
    on_actions = {
        correct_elf_mortality_trait 
    } 
}

correct_elf_mortality_trait = {
	effect = {
		# Fix ruler designer elf children
		every_living_character = {
			limit = {
				age < 20
				is_any_elf_immortal_type = yes
			}

			remove_elf_immortality_effect = yes

			debug_log = "fixed elf"
		}
	}
}
