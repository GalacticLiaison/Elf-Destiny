namespace = the_rut_events

the_rut_events.1 = {
    hidden = yes

    trigger = {
        age >= 16
        is_any_elf_type = yes
    }

    immediate = {
	    add_trait = rutting
    }
}

    hidden = yes

    trigger = {
        root = {
            has_trait = rutting
        }
    }

    immediate = {
        remove_trait = rutting
        add_character_flag = { flag = contraction_cooldown_rutting years = 8 }
    }
}
