		if = {
			limit = {
				exists = scope:match_recipient
			}

				limit = {
					scope:match_recipient = root
				}

					name = lord_or_heir
					value = flag:lord
				}
			}
			else_if = {
				limit = {
					scope:match_recipient = {
						is_heir_of = root
					}
				}
				save_scope_value_as = {
					name = lord_or_heir
					value = flag:heir
				}

		}

		}

				custom_tooltip = {
					text = aeluran_accepts_decision

			}

					add = 75
